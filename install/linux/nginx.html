# Nginx 的安装

  1. 安装工具包 wget、vim 和 gcc
1. yum install -y wget
2. yum install -y vim-enhanced
3. yum install -y make cmake gcc gcc-c++
1
2
3
  1. 下载 nginx 安装包
wget http://nginx.org/download/nginx-1.14.0.tar.gz
1
  1. 安装依赖包
1. yum install -y pcre pcre-devel
2. yum install -y zlib zlib-devel
3. yum install -y openssl openssl-devel
1
2
3
  1. 解压
tar -zxvf nginx-1.14.0.tar.gz -C /usr/local/
1
  1. configure 配置
进入nginx-1.14.0目录然后在执行./configure命令

[root@xxx nginx-1.14.0]# ./configure --prefix=/usr/local/nginx
1
2
3
  1. 编译安装
[root@xxx nginx-1.14.0]# make && make install
1
  1. 启动 Nginx，启动完之后检查 nginx 是否已经正常启动，看到如下信息说明正常启动
[root@xxx nginx-1.14.0]# /usr/local/nginx/sbin/nginx
[root@xxx nginx-1.14.0]# ps -ef | grep nginx
root     24956     1  0 19:41 ?        00:00:00 nginx: master process /usr/local/nginx/sbin/nginx
nobody   24957 24956  0 19:41 ?        00:00:00 nginx: worker process
root     24959 10533  0 19:41 pts/0    00:00:00 grep --color=auto nginx
[root@xxx nginx-1.14.0]#
1
2
3
4
5
6

关闭 nginx 命令

[root@xxx nginx-1.14.0]# /usr/local/nginx/sbin/nginx -s stop
1

重新热启动 nginx

[root@xxx nginx-1.14.0]# /usr/local/nginx/sbin/nginx -s reload
1
  1. 配置防火墙,nginx 默认的端口是 80(防火墙已经开启)
[root@xxx nginx-1.14.0]# firewall-cmd --zone=public --add-port=80/tcp --permanent
success
[root@xxx nginx-1.14.0]# firewall-cmd --reload
success
[root@xxx nginx-1.14.0]#
1

4
5
  1. 测试 nginx
浏览器访问ip地址即可
