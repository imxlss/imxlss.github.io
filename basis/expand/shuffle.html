Array.prototype.shuffle = function() {
  let length = this.length;

  for (let i = length - 1; i > 0; i--) {
    let index = Math.floor(Math.random() * (i + 1));
    [this[index], this[i]] = [this[i], this[index]];
  }

  return this;
};
