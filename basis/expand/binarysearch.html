# 代码结构

// 查找特定数值
function binarySearch(arr, target) {
  let left = 0,
    right = arr.length - 1;

  while (left <= right) {
    const mid = Math.floor((left + right) / 2);

    if (arr[mid] === target) {
      return mid; // 查找成功
    } else if (arr[mid] < target) {
      left = mid + 1; // 缩小区间
    } else if (arr[mid] > target) {
      right = mid - 1; // 缩小区间
    }
  }

  return -1; // 查找失败
}
